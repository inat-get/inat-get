def [] operand
  case operand
  when INatGet::Data::DSL::Condition::Nothing
    INatGet::Data::DSL::ANYTHING
  when INatGet::Data::DSL::Condition::Anything
    INatGet::Data::DSL::NOTHING
  when INatGet::Data::DSL::Condition::NOT
    operand.operand
  else
    new(operand).freeze
  end
end
