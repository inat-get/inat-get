def to_sequel value
  cond = case value
  when ::Range
    { rank_level: (value.begin&.level .. value.end&.level) }
  when Enumerable
    { rank: value.map(&:to_s) }
  when INatGet::Data::Enum::Rank
    { rank: value.to_s }
  else
    { rank: value }
  end
  if self.helper.endpoint == :observations
    require_relative '../../models/taxon'
    { taxon_id: INatGet::Data::Model::Taxon.where(cond).select(:id) }
  else
    cond
  end
end
