def to_api value
  result = {}
  case @key
  when :observed
    result[:d1] = value.begin if value.begin
    result[:d2] = value.end   if value.end
  when :created
    result[:created_d1] = value.begin if value.begin
    result[:created_d2] = value.end   if value.end
  end
  result
end

end
