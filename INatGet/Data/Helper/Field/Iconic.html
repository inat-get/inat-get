def value_to_sequel value
  case value
  when Enumerable
    value.map { |v| value_to_sequel(v) }.to_set
  when INatGet::Data::Enum::Iconic
    value.taxon_id
  when String
    INatGet::Data::Enum::Iconic::parse(value).taxon_id
  else
    value
  end
end
