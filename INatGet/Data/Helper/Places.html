def query_to_api **query
  endpoint = self.manager.endpoint
  values = (query[:id] || []).to_set + (query[:slug] || []).to_set
  blocks = values.each_slice(200).to_a
  blocks.map { |v| { endpoint: "#{ endpoint }/#{ v.map(&:to_s).join(',') }", query: {} } }
end
